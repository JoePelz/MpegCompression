public class MyNode : Node {

	public MyNode() {
		//default setup
	}

	protected override void createProperties() {
        Property p = properties["name"];
        //set custom name
        p.setString("MyNode");
	}

	protected override void createInputs() {
		inputs.Add("myInput", null);
    }

    protected override void createOutputs() {
        outputs.Add("myOutput", new HashSet<Address>());
    }

	public override DataBlob getData(string port) {
		base.getData(port);
        DataBlob d = new DataBlob();
        d.type = DataBlob.Type.*;
        d.* = myData;
        return d;
    }

	protected override void clean() {
        base.clean();
		//read inputs and update output...
	}
}
