public class MyNode : Node {

	public MyNode() {
		//default setup
	}

	protected override void createProperties() {
        base.createProperties();
        Property p = properties["name"];
        //set custom name
        p.setString("MyNode");
	}

	protected override void createInputs() {
        //no inputs for ReadImage. Only string property.
		inputs.Add("myInput", null);
    }

    protected override void createOutputs() {
        outputs.Add("myOutput", new HashSet<Address>());
    }

	protected override DataBlob getData(string port) {
		base.getData(port);
        DataBlob d = new DataBlob();
        d.type = DataBlob.Type.*;
        d.* = myData;
        return d;
    }

	protected override void clean() {
        base.clean();
		//read inputs and update output...
	}

	public override Bitmap view() {
        base.view();
        if (data == null) {
            return;
        }
        //create bitmap from data
		return bmp;
	}
	
    public override Rectangle getExtents() {
        return new Rectangle(-bmp.Width / 2, -bmp.Height / 2, bmp.Width, bmp.Height);
    }
}
